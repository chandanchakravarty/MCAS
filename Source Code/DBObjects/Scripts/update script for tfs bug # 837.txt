IF EXISTS (SELECT * FROM ACT_GL_ACCOUNTS where ACC_LEVEL_TYPE='AS')
BEGIN
      DECLARE @LOOKUP_UNIQUE_ID INT
            SELECT @LOOKUP_UNIQUE_ID=LOOKUP_UNIQUE_ID FROM MNT_LOOKUP_VALUES WHERE LOOKUP_VALUE_CODE='AS' AND LOOKUP_ID=1333
            Update ACT_GL_ACCOUNTS Set ACC_LEVEL_TYPE=@LOOKUP_UNIQUE_ID WHERE ACC_LEVEL_TYPE='AS' 
END
GO
